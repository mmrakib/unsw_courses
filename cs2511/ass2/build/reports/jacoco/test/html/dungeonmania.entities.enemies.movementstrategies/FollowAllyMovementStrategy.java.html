package dungeonmania.entities.enemies.movementstrategies;

import dungeonmania.Game;
import dungeonmania.entities.Player;
import dungeonmania.entities.enemies.Enemy;
import dungeonmania.entities.enemies.Mercenary;
import dungeonmania.map.GameMap;
import dungeonmania.util.Position;

public class FollowAllyMovementStrategy implements MovementStrategy {
    @Override
    public Position getNextMove(Game game, Enemy enemy) {
        Position nextPos;
        GameMap map = game.getMap();
        Player player = game.getPlayer();
        Mercenary merc = (Mercenary) enemy;

        boolean isAdjacentToPlayer = merc.getIsAdjacentToPlayer();
        nextPos = isAdjacentToPlayer ? player.getPreviousDistinctPosition()
                : map.dijkstraPathFind(enemy.getPosition(), player.getPosition(), enemy);
        if (!isAdjacentToPlayer && Position.isAdjacent(player.getPosition(), nextPos))
            isAdjacentToPlayer = true;

        return nextPos;
    }
}
